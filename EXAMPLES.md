from PyQt6.QtGui import QColor

def custom_colormap(value, max_value):
    """
    –°–æ–∑–¥–∞–µ—Ç RGB —Ü–≤–µ—Ç –Ω–∞ –æ—Å–Ω–æ–≤–µ –∑–Ω–∞—á–µ–Ω–∏—è –∏—Ç–µ—Ä–∞—Ü–∏–∏
    
    Args:
        value: –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π (0 to max_value)
        max_value: –ú–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π
    
    Returns:
        Tuple (r, g, b)
    """

        return (0, 0, 0)  # –ß–µ—Ä–Ω—ã–π –¥–ª—è —Ç–æ—á–µ–∫ –≤ –º–Ω–æ–∂–µ—Å—Ç–≤–µ
    
    # –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è –∑–Ω–∞—á–µ–Ω–∏—è

    # –ü–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å—Å–∫–∏–π –≥—Ä–∞–¥–∏–µ–Ω—Ç
    r = int(255 * np.sin(t * np.pi * 2))
    g = int(255 * np.cos(t * np.pi * 3))
    b = int(255 * np.sin(t * np.pi * 4 + np.pi/2))
    
    return (abs(r), abs(g), abs(b))
